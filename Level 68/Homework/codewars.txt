function invert(arr) {
  let res = []; // ვქმნი ცარიელ მასივს, სადაც შევინახავ შედეგს
  for(let i = 0; i < arr.length; i++){ // ვუყვები მასივს ელემენტების რაოდენობით
    res.push(-arr[i]); // ყოველ ელემენტს ვაკეთებ უარყოფითს და ვამატებ ახალ მასივში
  }
  return res; // ბოლოს ბრუნვავ ყველა შედეგს
}
# 2
function getAverage(marks) {
  let sum = 0; // აქ ვქმნით ცვლადს, სადაც შევინახავთ ყველა ნიშნის ჯამს
  for (let i = 0; i < marks.length; i++) { // ეს ციკლი მიმოიხილავს ყველა ნიშანს მასივში
    sum += marks[i]; // ყველა ნიშანს ვუმატებთ ჯამს
  }
  return Math.floor(sum / marks.length); // ვიღებთ საშუალოს და ზედა ნაწილს ვჭრით
}

function check(a, x) {
  let found = false; // დავიწყებთ იმით, რომ არ ვიცით, არის თუ არა x მასივში
  for (let i = 0; i < a.length; i++) { // ციკლი გადავა ყველა ელემენტზე
    if (a[i] === x) { // თუ ვიპოვეთ x, მაშინ found გახდება true
      found = true;
    }
  }
  return found; // ვაბრუნებთ true-ს, თუ x მოიძებნა, და false-ს, თუ არა
}
